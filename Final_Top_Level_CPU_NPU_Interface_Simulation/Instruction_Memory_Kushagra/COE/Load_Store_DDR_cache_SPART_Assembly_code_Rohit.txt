# R3 - Different Data written in memory

# R6 - 
LLW R2 0x0008 ;

LLW R6 0x0000 ;
LHW R6 0x0800 ;
LLW R7 0x0001 ;
LHW R7 0x0800 ;
LLW R8 0x0002 ;
LLW R9 0x0001 ;



SPART_RX_NOT_READY_1 LOAD R10 R7 0x00 ;
AND R0 R10 R8 ;
B EQ SPART_RX_NOT_READY_1 ;

LOAD R15 R6 0x00 ;

ADD R0 R0 R0 ;
ADD R0 R0 R0 ;

STORE R15 R2 ;

ADD R2 R2 R9 ;

SUB R4 R4 R9 ;
