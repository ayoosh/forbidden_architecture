
# Set Stack Pointer ;
LLW R1 0XFFFF ;
LHW R1 0X03FF ;

# Infinitely send STACK POINTER value via SPART
# R2 - SPART Address
# R3 -  SPART Status address
# R4 - SPART Read Mask
# R5 - SPART Write Mask
# R6 - Value read from spart
ADD R2 R0 R0 ;
LHW R2 0x0800 ;
LLW R3 0x0001 ;
LHW R3 0x0800 ;
LLW R4 0x0002 ;
LLW R5 0x0001 ;

SPART_TX_NOT_READY_1 LOAD R6 R3 0x00 ;
AND R0 R6 R5 ;
B EQ SPART_TX_NOT_READY_1 ;

STORE R1 R2 ;
B UNCOND SPART_TX_NOT_READY_1 ;
